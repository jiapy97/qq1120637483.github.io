var kthLargest = function(root, k) {

    const set = new Set();
    const dfs = function(node) {
        if (node === null) {
            return;
        }

        set.add(node.val);
        dfs(node.left);
        dfs(node.right);
    }
    dfs(root);
    const arr = [...set];
    arr.sort((a,b) => {
        return b-a;
    });

    return arr[k-1];

};
