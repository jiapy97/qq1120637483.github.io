  1. 附带条件的查询

使用py2neo连接数据库

from py2neo import Graph,Node,Relationship,Subgraph
from py2neo.matching import *


graph = Graph("bolt://localhost:7687", auth=("neo4j", "1234"))
success = graph.run("UNWIND range(1, 3) AS n RETURN n, n * n as n_sq")
if (success):
  print("连接成功")
