function titleCase(str) {
    const result = str.split(' ');
    for (let i in result) {
        result[i] = result[i][0].toUpperCase() + result[i].slice(1,result[i].length).toLowerCase(); 
    }
    result
    const test = result.reduce((pre,value) => {
        return pre + ' ' + value;
    })
    
    return test;
}

剑指Offer——和为S的两个数字（JS实现）

2021年4月16日 · 阅读需 1 分钟

题目描述​

解题思路​

  • 使用双指针
  • 一个指针指向最左端
  • 一个指针指向最右端
  • 大了最右端的往左移
  • 小了最左端的往右移

实现代码​

var twoSum = function (nums, target) {
    let l = 0;
    let r = nums.length-1;
    while (nums[l] + nums[r] !== target) {
        if (nums[l] + nums[r] < target) {
            l++;
        }
        if (nums[l] + nums[r] > target) {
            r--;
        }
    }
    return [nums[l],nums[r]]

};

剑指Offer——数组中出现次数超过一半的数字（JS实现）

2021年4月16日 · 阅读需 1 分钟

题目描述​

解题思路​

  • 使用哈希Map的键存放数组的元素
  • 使用哈希Map的值存放该元素出现的次数
  • 找出出现次数大于长度一半的元素，返回即可

实现代码​

var majorityElement = function(nums) {
    // 存储数组的长度的一半
    let len = nums.length/2;
    const m = new Map();
    for (let v of nums) {
        if (m.has(v)) {
            m.set(v,m.get(v)+1);
        } else {
            m.set(v,1);
        }
    };
    for (let v of m) {
        if (v[1] > len) {
            return v[0];
        }
    }
};

解题思路​

  • 只要出现重复，直接返回

实现代码​

var findRepeatNumber = function(nums) {
    const m = new Map();
    for (let v of nums) {
        if (m.has(v)) {
            return v;
        } else {
            m.set(v,1);
        }
    }
};
