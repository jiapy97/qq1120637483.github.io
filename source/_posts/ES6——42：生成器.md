## 生成器函数实例
```js
// 1s后控制台输出111 2s后控制台输出222 3s后控制台输出333
function one() {
    setTimeout(() => {
        console.log(111);
        iterator.next();
    },1000);
}

function two() {
    setTimeout(() => {
        console.log(222);
        iterator.next();
    },2000);
}

function three() {
    setTimeout(() => {
        console.log(333);
        iterator.next();
    },3000);
}

function * gen() {
    yield one();
    yield two();
    yield three();
}

// 获取迭代器对象
const iterator = gen();
iterator.next();
```
