var treeToDoublyList = function (root) {
    const temp = [];
    function dfs(root) {
        if (!root) return null;

        dfs(root.left);
        temp.push(root)
        dfs(root.right);
        return null;
    }
    dfs(root);
    const len = temp.length;
    for (let i = 0; i < temp.length; i++) {
        if (i !== 0) {
            temp[i].left = temp[i - 1];
        } else {
            temp[i].left = temp[len-1];
        }
        if (i !== len - 1) {
            temp[i].right = temp[i + 1];
        } else {
            temp[i].right = temp[0];
        }
    }
    return temp[0];
};
